70void Engine::handleEvents() {
71 sf::Event event;
72 while (m_window.pollEvent(event)) {
73 if (event.type == sf::Event::Closed)
74 m_window.close();
75 if (event.type == sf::Event::Resized) {
76
77 // Update engine info
78 m_resoltuon.setX(event.size.width);
79 m_resoltuon.setY(event.size.height);
80 // make new view
81 sf::FloatRect view{0, 0, static_cast<float>(m_resoltuon.getX()),
82 static_cast<float>(m_resoltuon.getY())};
83 m_window.setView(sf::View{view});
84
85 LOGINFO << "Resized\t" << event.size.width << '\t' << event.size.height
86 << '\n';
87 }
88
89 // Fire custom event handler.
90 m_eventHandlers[event.type](event);
91 }
